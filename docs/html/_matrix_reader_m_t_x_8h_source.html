53 void parseFile(const std::string &filename)
54 {

80 void processHeader(std::ifstream &file)
81 {
82 // Get first line and look for "complex" or "symmetric"
83 std::string line;
84 std::getline(file, line);
85
86 isComplex = false;
87 isSymmetric = false;
88 if (line.find("complex") != std::string::npos)
89 {
90 isComplex = true;
91 }
92 if (line.find("symmetric") != std::string::npos)
93 {
94 isSymmetric = true;
95 }
96
97 // Skip rest of lines starting with %
98 while (line[0] == '%')
99 {
100 std::getline(file, line);
101 }
102
103 // Get dimensions (first line after % comments)
104 std::stringstream ss(line);
105 ss >> rows >> cols >> nnz;
106 }

108 void readMatrix(std::ifstream &file)
109 {
110 matrix = Eigen::SparseMatrix<Scalar>(rows, cols);
111
112 // Read each line
113 std::string line;
114 while (std::getline(file, line))
115 {
116 // Skip comments
117 if (line[0] == '%')
118 {
119 continue;
120 }
121
122 // Parse line
123 std::stringstream ss(line);
124 int row, col;
125 double real, imag;
126
127 if (isComplex)
128 {
129 ss >> row >> col >> real >> imag;
130 addComplexEntry(row, col, real, imag);
131 }
132 else
133 {
134 ss >> row >> col >> real;
135 addRealEntry(row, col, real);
136 }
137 }
138 }
