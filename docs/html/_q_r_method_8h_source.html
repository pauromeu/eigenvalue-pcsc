170 Eigen::VectorX<std::complex<double>> computeComplexConjugateEigenvalues(const Eigen::MatrixX<Scalar> &block)
171 {
172 // Compute the coefficients of the quadratic equation
173 Scalar a = 1; // Coefficient of λ^2
174 Scalar b = -(block(0, 0) + block(1, 1)); // Coefficient of λ
175 Scalar c = block(0, 0) * block(1, 1) - block(0, 1) * block(1, 0); // Constant term
176
177 // Compute the discriminant
178 Scalar discriminant = b * b - Scalar(4) * a * c;
179
180 Eigen::VectorX<std::complex<double>> roots(2);
181
182 // Check if Scalar is a complex type
183 if constexpr (std::is_same<Scalar, std::complex<float>>::value || std::is_same<Scalar, std::complex<double>>::value)
184 {
185 Scalar sqrtDiscriminant = std::sqrt(discriminant);
186 roots[0] = std::complex<double>((-b + sqrtDiscriminant) / (Scalar(2) * a));
187 roots[1] = std::complex<double>((-b - sqrtDiscriminant) / (Scalar(2) * a));
188 }
189 else // Handle real number types
190 {
191 std::complex<Scalar> complexDiscriminant = discriminant;
192 std::complex<Scalar> sqrtDiscriminant = std::sqrt(complexDiscriminant);
193 roots[0] = std::complex<double>((-b + sqrtDiscriminant) / (Scalar(2) * a));
194 roots[1] = std::complex<double>((-b - sqrtDiscriminant) / (Scalar(2) * a));
195 }
